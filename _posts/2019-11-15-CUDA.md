> 完成cuda的“Hello world”程序，运行参数`grid(2,4), block(8,16)`，给出输出结果文件。

下为`HelloWorld.cu`。

```clike
#include <stdio.h>
__global__ void kernel()
{
	printf("Hello world from block(%d,%d,%d) thread(%d,%d,%d).\n",
		   blockIdx.x, blockIdx.y, blockIdx.z,
		   threadIdx.x, threadIdx.y, threadIdx.z);
}
int main()
{
	dim3 grid(2, 4), block(8, 16);
	kernel<<<grid, block>>>();
	cudaDeviceSynchronize();
}
```

编译并运行。

```bash
nvcc HelloWorld.cu -o HelloWorld
./HelloWorld > HelloWorld.txt
```

> 完成CUDA的两个矩阵加法A+B=C，其中A、B是$2^{13}\times 2^{13}$的方阵。假设矩阵A的元素为`a[i][j]=i-0.1*j+1`，矩阵B的元素为`b[i][j]=0.2*j-0.1*i`。

下为`MatMatAdd.cu`。其实矩阵加法可以直接类比于向量加法来做。

```clike
#include <stdlib.h>
typedef double lf;
__global__ void MatAdd(lf *A, const lf *B, const int nRow, const int nCol)
{
	int i = blockIdx.x * blockDim.x + threadIdx.x,
		j = blockIdx.y * blockDim.y + threadIdx.y;
	if (i < nRow && j < nCol)
		A[i * nCol + j] += B[i * nCol + j];
}
int main()
{
	int nRow = 1 << 13,
		nCol = 1 << 13,
		nElem = nRow * nCol,
		nBytes = nElem * sizeof(lf);
	lf *h_A = (lf *)malloc(nBytes),
	   *h_B = (lf *)malloc(nBytes),
	   *d_A, *d_B;
	for (int i = 0; i < nRow; ++i)
		for (int j = 0; j < nCol; ++j)
		{
			h_A[i * nCol + j] = i - 0.1 * j + 1;
			h_B[i * nCol + j] = 0.2 * j - 0.1 * i;
		}
	cudaMalloc((lf **)&d_A, nBytes);
	cudaMalloc((lf **)&d_B, nBytes);
	cudaMemcpy(d_A, h_A, nBytes, cudaMemcpyHostToDevice);
	cudaMemcpy(d_B, h_B, nBytes, cudaMemcpyHostToDevice);
	dim3
		threadsPerBlock(32, 32),
		numBlocks(nRow / threadsPerBlock.x, nCol / threadsPerBlock.y);
	MatAdd<<<numBlocks, threadsPerBlock>>>(d_A, d_B, nRow, nCol);
	cudaMemcpy(h_A, d_A, nBytes, cudaMemcpyDeviceToHost);
	cudaFree(d_A);
	cudaFree(d_B);
	free(h_A);
	free(h_B);
}
```

```bash
nvcc MatMatAdd.cu -o MatMatAdd
./MatMatAdd
```
